% 01_packages.tex
% Paquetes esenciales para el cumplimiento de ISO/IEC/IEEE 26514:2022

% Codificación e idioma
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage[spanish,es-tabla]{babel} % Soporte en español

% Familias Tipográficas (Norma de Diseño)
\usepackage{mathptmx}  % Establece Serif (Times) como fuente principal para el cuerpo
\usepackage{helvet}    % Establece Sans-serif (Helvetica) para la familia sin remates
\usepackage{courier}   % Establece Monospace (Courier) para la familia monoespaciada

% Geometría y Layout
\usepackage{geometry}  % Control preciso de los márgenes de página
\usepackage{setspace}  % Configuración de interlineado (1.5 requerido)
\usepackage{fancyhdr}  % Personalización de encabezados y pies de página
\usepackage{lastpage}  % Permite referenciar el total de páginas (Ej. Página X de Y)
\usepackage{titlesec}  % Formateo de títulos y subtítulos (tamaño y estilo)

% Gráficos, Colores y Símbolos
\usepackage{graphicx}      % Inserción de imágenes y logos corporativos
\usepackage{xcolor}        % Definición de paleta de colores corporativa
\usepackage{fontawesome5}  % Iconografía para componentes de avisos (Nota, Precaución)
\usepackage{amssymb}       % Símbolos adicionales (ej. cuadros para checklists)

% Cajas de Diseño y Listas
\usepackage[most]{tcolorbox} % Creación de cajas visuales para avisos y ayudas
\usepackage{enumitem}        % Personalización de viñetas para checklists

% Navegación y Metadatos PDF
\usepackage{hyperref}  % Generación de enlaces internos y propiedades del documento
