% 1. Configuración de Márgenes Exactos (Corrección ISO Definitiva)
\geometry{
    a4paper,
    inner=3cm,       % Margen izquierdo para encuadernación
    outer=2cm,       % Margen derecho
    top=2.5cm,       % Distancia exacta del borde superior de la hoja al ENCABEZADO

    includehead,     % Instrucción clave: El encabezado va DENTRO del margen de 2.5cm
    includefoot,     % Instrucción clave: El pie de página va DENTRO del margen de 2.5cm

    headsep=0.2cm,   % Separación estética entre la línea del encabezado y el texto
    footskip=1.0cm   % Separación estética entre el final del texto y la línea del pie
}
